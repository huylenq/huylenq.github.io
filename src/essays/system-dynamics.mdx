export const stockValue = (get) => {
  const inflow = get("inflow");
  const outflow = get("outflow");
  const dt = get("dt");
  const net = (inflow - outflow) * dt;
  return net >= 0 ? `+${net.toFixed(1)}` : net.toFixed(1);
};

export const afterSteps = (get) => {
  const inflow = get("inflow");
  const outflow = get("outflow");
  const dt = get("dt");
  const initial = 100;
  const steps = 10;
  let stock = initial;
  for (let i = 0; i < steps; i++) {
    stock += (inflow - outflow) * dt;
  }
  return stock.toFixed(0);
};

export const behavior = (get) => {
  const inflow = get("inflow");
  const outflow = get("outflow");
  if (inflow > outflow) return "growing";
  if (inflow < outflow) return "draining";
  return "in equilibrium";
};

export const delayOsc = (get) => {
  const delay = get("delay");
  if (delay <= 1) return "almost none — fast correction";
  if (delay <= 3) return "mild wobble around target";
  if (delay <= 6) return "noticeable oscillation";
  return "wild swings — the system hunts badly";
};

export const switchingPenalty = (get) => {
  const wip = get("wip");
  const devs = get("devs");
  const perDev = wip / devs;
  const penalty = Math.min(0.8, Math.max(0, (perDev - 2) * 0.2));
  return (penalty * 100).toFixed(0);
};

export const debtDrag = (get) => {
  const debt = get("debt");
  return Math.min(90, debt * 5).toFixed(0);
};

export const velocity = (get) => {
  const devs = get("devs");
  const wip = get("wip");
  const debt = get("debt");
  const baseVelocity = devs * 3; // 3 story points per dev per sprint
  const perDev = wip / devs;
  const switchLoss = Math.min(0.8, Math.max(0, (perDev - 2) * 0.2));
  const debtLoss = Math.min(0.9, debt * 0.05);
  return Math.max(0, baseVelocity * (1 - switchLoss) * (1 - debtLoss)).toFixed(1);
};

export const bugRate = (get) => {
  const wip = get("wip");
  const debt = get("debt");
  // More concurrent work + more debt = more bugs
  return (wip * 0.05 * (1 + debt * 0.15)).toFixed(1);
};

export const netThroughput = (get) => {
  const devs = get("devs");
  const wip = get("wip");
  const debt = get("debt");
  const bugs = get("bugs");
  const baseVelocity = devs * 3;
  const perDev = wip / devs;
  const switchLoss = Math.min(0.8, Math.max(0, (perDev - 2) * 0.2));
  const debtLoss = Math.min(0.9, debt * 0.05);
  const vel = Math.max(0, baseVelocity * (1 - switchLoss) * (1 - debtLoss));
  // Bug fixing eats into velocity: each bug costs ~0.5 points to fix
  const bugTax = bugs * 0.5;
  return Math.max(0, vel - bugTax).toFixed(1);
};

export const sprintSim = (get) => {
  const devs = get("devs");
  let wip = get("wip");
  let debt = get("debt");
  let bugs = get("bugs");
  let backlog = get("backlog");
  let totalShipped = 0;

  for (let sprint = 0; sprint < 6; sprint++) {
    const baseVel = devs * 3;
    const perDev = wip / Math.max(1, devs);
    const switchLoss = Math.min(0.8, Math.max(0, (perDev - 2) * 0.2));
    const debtLoss = Math.min(0.9, debt * 0.05);
    const vel = Math.max(0, baseVel * (1 - switchLoss) * (1 - debtLoss));

    // New bugs generated this sprint
    const newBugs = wip * 0.05 * (1 + debt * 0.15);
    bugs = Math.max(0, bugs + newBugs);

    // Devs split effort: fix bugs first, then ship features
    const bugFixCapacity = Math.min(bugs, vel * 0.4); // max 40% on bugs
    const featureCapacity = vel - bugFixCapacity;
    const shipped = Math.min(wip, featureCapacity);

    bugs = Math.max(0, bugs - bugFixCapacity);
    wip = Math.max(0, wip - shipped);
    totalShipped += shipped;

    // Pull new work from backlog
    const pull = Math.min(backlog, devs * 2);
    wip += pull;
    backlog = Math.max(0, backlog - pull);

    // Debt accrues slowly when WIP is high (rushing → shortcuts)
    if (perDev > 3) debt = Math.min(18, debt + 0.5);
  }

  return totalShipped.toFixed(0);
};

export const dominantLoop = (get) => {
  const wip = get("wip");
  const devs = get("devs");
  const debt = get("debt");
  const bugs = get("bugs");
  const perDev = wip / devs;

  if (debt >= 12 && bugs >= 10) return "death spiral — bugs and debt reinforce each other, velocity collapses";
  if (perDev > 4 && debt >= 8) return "reinforcing (overload → debt → bugs → more overload)";
  if (bugs >= 8) return "balancing (bug-fixing absorbs most capacity)";
  if (perDev > 3) return "transitional (context-switching starting to bite)";
  return "healthy — reinforcing loop dormant, team ships steadily";
};

## The Computational Model of Systems

Systems theory isn't philosophy — it's a modeling paradigm that maps directly to simulation code. Donella Meadows' core insight: *system structure is the source of system behavior.* Capture structure correctly and behavior emerges on its own. No special cases. No hand-tuning.

## Stock-Flow: The Fundamental Primitive

Every system dynamics model reduces to three ideas: **stocks** accumulate, **flows** change them, **feedback** connects the two. The update rule is just Euler integration:

```
stock[t+1] = stock[t] + (inflow - outflow) × dt
```

## The Three Building Blocks

**Stocks** are the system's memory. Population, inventory, bank balance, knowledge — anything that accumulates. They only change through flows, never instantaneously. This constraint is what gives systems their inertia, and inertia is what makes systems *interesting*.

**Flows** control how fast stocks change. Birth rates, sales rates, learning rates. They can be constant, computed from other stocks, or driven by policy decisions.

**Feedback loops** are where the magic lives. A balancing loop seeks a goal (a thermostat). A reinforcing loop amplifies (compound interest). The Sufi teaching story Meadows loved: you cannot understand "one" or "two" without understanding "and" — feedback is the *and* between stocks.

## Delays Make Systems Oscillate

A delay in a balancing feedback loop makes a system oscillate. The classic example: you're in a shower, the water is cold, you crank it hot. But there's a pipe delay. By the time hot water arrives, you've overcorrected. Now it's scalding. You crank it back...

Computationally, delays are modeled as exponential smoothing — the perceived value chases the actual value:

```
perceived[t+1] = perceived[t] + (actual[t] - perceived[t]) / delay × dt
```

The feedback signal is always slightly stale. This is why systems "hunt" around their target, and why supply chains bullwhip, and why monetary policy lags.

## Nonlinearity: An Agile Team as a Dynamic System

Forget biology. Here's a system every software engineer has *lived inside*: an agile team. The stocks are familiar — **backlog**, **work-in-progress**, **bugs**, **tech debt** — but few people model the feedback loops that connect them. Let's fix that.

### The Stocks

```
Backlog     →  tickets waiting to be pulled
WIP         →  tickets actively being worked
Bugs        →  known defects awaiting fixes
Tech Debt   →  accumulated shortcuts (0 = pristine, 18 = legacy horror)
```

### The Flows and Feedback

The interesting dynamics emerge from three interacting loops:

**Context-switching penalty** (balancing): When each dev juggles too many tickets, cognitive overhead eats velocity. Below about 2 tickets per dev, negligible cost. Above that, every additional ticket makes *all* tickets slower.

**Tech debt drag** (reinforcing): Debt makes code harder to change → features take longer → pressure to cut corners → more debt. This loop is dormant when debt is low and devastating when it compounds.

**Bug generation** (reinforcing): More concurrent WIP in crufty code produces more bugs. More bugs steal capacity from features. Falling behind on features → pressure to rush → more debt → more bugs. This is the death spiral.

### What the Model Reveals

Try these experiments:

**The "just add more WIP" trap.** Set devs to 5, debt to 2, bugs to 2. Start with WIP at 8 — things look healthy. Now crank WIP to 20. Watch context-switching eat velocity alive. Gross throughput *drops* even though more work is "in progress." This is Little's Law fighting back: throughput = WIP / cycle time, and cycle time is exploding.

**The debt spiral.** Set WIP to 10, bugs to 2, then slide debt from 2 to 14. Velocity collapses. Bug generation rate climbs because defects breed faster in messy code. Net throughput approaches zero. This is the codebase where every sprint is a slog and the team "doesn't understand why we're so slow."

**The recovery.** From that grim state (debt 14, bugs 12, WIP 15), try the counter-intuitive fix: *reduce* WIP to 5 and hold. Context-switching drops. Fewer concurrent changes means fewer new bugs. The team can start paying down debt instead of generating it. Six-sprint throughput actually *rises* with less WIP.

This is the core lesson of system dynamics applied to software: the reinforcing loops between WIP, bugs, and tech debt mean that **pushing harder makes things worse** past a threshold. The only way out is to reduce WIP, which feels like slowing down but is structurally the opposite.

## Bounded Rationality

This is also why swapping out the engineering manager, or the "slow" developer, rarely fixes a struggling team. When simulating human systems, agents don't optimize — they *satisfice* based on delayed, filtered information. The beer game proves this every semester: smart MBA students produce the same wild oscillations because the *structure* (delays, incomplete information, local incentives) dictates behavior, not individual intelligence.

The PM sees the backlog growing and pushes more tickets into WIP. The tech lead sees bugs piling up and diverts developers to firefighting. The CTO sees velocity declining and asks for status reports, adding overhead. Everyone is locally rational. The system is globally dysfunctional. Replacing any one of them changes nothing — the feedback structure remains.

## Validating Models

System dynamics models aren't point predictors. You validate them by checking for archetypal behavior modes under appropriate conditions:

- **Exponential growth/decay** — crank debt and watch bugs compound
- **Goal-seeking** — set WIP below capacity and watch throughput stabilize
- **Oscillation** — add feedback delays (not modeled here, but imagine sprint retrospectives lagging by 2 sprints)
- **S-shaped growth** — a new team ramps up fast, then plateaus as coordination costs grow
- **Overshoot and collapse** — WIP spike → debt accumulation → velocity crash

If the model produces these patterns under the right parameter regimes, it has structural validity. That's what matters — not whether it predicts next sprint's velocity.
